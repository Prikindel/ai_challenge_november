    private val textChunker: TextChunker,
    private val embeddingService: EmbeddingService,
    private val vectorNormalizer: VectorNormalizer,
    private val knowledgeBaseRepository: KnowledgeBaseRepository

    suspend fun indexDocument(filePath: String): IndexingResult {
        // 1. –ó–∞–≥—Ä—É–∑–∫–∞ –¥–æ–∫—É–º–µ–Ω—Ç–∞

            val normalized = vectorNormalizer.normalizeTo01(embedding)

                id = generateId(),
                documentId = document.id,
                chunkIndex = chunk.chunkIndex,
                content = chunk.content,
                startIndex = chunk.startIndex,
                endIndex = chunk.endIndex,
                tokenCount = chunk.tokenCount,
                embedding = normalized
            )
        }

        knowledgeBaseRepository.saveChunks(indexedChunks)
        
        return IndexingResult(
            documentId = document.id,
            chunksCount = indexedChunks.size,
            success = true
        )
    }
}

class CosineSimilarityCalculator {
    fun calculateSimilarity(
        vector1: List<Float>,
        vector2: List<Float>
    ): Float {
        require(vector1.size == vector2.size) {
            "Vectors must have the same size"
        }
        
        val dotProduct = vector1.zip(vector2).sumOf { (a, b) -> a * b }
        val magnitude1 = sqrt(vector1.sumOf { it * it }.toDouble())
        val magnitude2 = sqrt(vector2.sumOf { it * it }.toDouble())
        
        return (dotProduct / (magnitude1 * magnitude2)).toFloat()
    }
}

class KnowledgeBaseSearchService(

    private val knowledgeBaseRepository: KnowledgeBaseRepository,
    private val similarityCalculator: CosineSimilarityCalculator
